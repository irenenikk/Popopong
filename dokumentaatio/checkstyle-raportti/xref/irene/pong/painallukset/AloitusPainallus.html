1   package irene.pong.painallukset;
2   
3   import irene.pong.kayttoliittyma.Animoija;
4   import irene.pong.kayttoliittyma.Kayttoliittyma;
5   import irene.pong.logiikka.Pong;
6   import java.awt.event.ActionEvent;
7   import javax.swing.AbstractAction;
8   
9   /**
10   * Luokka sitoo välilyönnin painamiseen pelin aloittamisen.
11   */
12  public class AloitusPainallus extends AbstractAction {
13  
14      private final Pong pong;
15      private final Kayttoliittyma kali;
16  
17      /**
18       * Luokka saa konstruktorissaan sekä päälogiikkaluokan, että pääkäyttöliittymäluokan.
19       * 
20       * @param p Pelin päälogiikkaluokka
21       * @param k Pelin pääkäyttöliittymäluokka
22       */
23      public AloitusPainallus(Pong p, Kayttoliittyma k) {
24          pong = p;
25          kali = k;
26      }
27  
28      @Override
29      public void actionPerformed(ActionEvent e) {
30          /**
31           * Jos peli on vasta aloitettu, päivittää päälogiikkaluokan tilan, tyhjentää kentän yläreunan ilmoituksen ja aloittaa pelianimaation.
32           * Jos peli on juuri loppunut, nollaa pelitilanteen nollaamalla tilastoluokan pisteetsekä mailoille viimeksi annetut suunnat, poistaa tekoälyn käytöstä, ja palauttaa pallon nopeuden aloitusnopeudeksi.
33           */
34          if (!pong.isPeliKaynnissa() && !kali.getAnimoija().getAnimaatio().isPaused() && !pong.isPeliAlkanut() && pong.getPelityyppi() == null) {
35              pong.getTilasto().nollaaPisteet();
36              pong.getKentta().alustaKentta();
37              kali.getPiirturi().tyhjennaEsteidenVarit();
38              pong.getKontrolleri().alustaKomponentit();
39              kali.pausePiiloon();
40              kali.paivita();
41              kali.getAnimoija().getAnimaatio().stop();
42  
43          }
44          if (!pong.isPeliKaynnissa() && !kali.getAnimoija().getAnimaatio().isPaused() && !pong.isPeliAlkanut() && pong.getPelityyppi() != null) {
45              pong.setPeliKaynnissa(true);
46              pong.setPeliAlkanut(true);
47              kali.getAnimoija().aloitaPeli();
48              kali.pauseNakyvaksi();
49              kali.tyhjennaIlmoitus();
50          }
51      }
52  
53  }
