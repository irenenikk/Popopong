5   package irene.pong.painallukset;
6   
7   import irene.pong.kayttoliittyma.Animoija;
8   import irene.pong.kayttoliittyma.Kayttoliittyma;
9   import irene.pong.logiikka.Pong;
10  import java.awt.event.ActionEvent;
11  import javax.swing.AbstractAction;
12  
13  public class PauseNappi extends AbstractAction {
14  
15      private final Pong pong;
16      private final Kayttoliittyma kali;
17      
18      public PauseNappi(Pong p, Kayttoliittyma i) {
19          pong = p;
20          kali = i;
21      }
22  
23      @Override
24      public void actionPerformed(ActionEvent e) {
25          if (pong.peliAlkanut()) {
26              if (pong.isPeliKaynnissa()) {
27                  pong.setPeliKaynnissa(false);
28                  kali.asetaIlmoitus("paused");
29                  kali.pausePiiloon();
30                  kali.getAnimoija().pauseta();
31              } else {
32                  pong.setPeliKaynnissa(true);
33                  kali.tyhjennaIlmoitus();
34                  kali.pauseNakyvaksi();
35                  kali.getAnimoija().jatka();
36              }            
37          }
38      }
39  }
