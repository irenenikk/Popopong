1   /**
2    * Pelin tarvittavat osat luova luokka, joka myös säilyttää tiedon käyttäjän valitsemasta pelityypistä.
3    * Asettaa näppäimet kuunneltaviksi ja aloittaa pelin.
4    */
5   package irene.pong.logiikka;
6   
7   import irene.pong.grafiikka.Animoija;
8   import irene.pong.grafiikka.Ikkuna;
9   import irene.pong.kayttoliittyma.AloitusPainallus;
10  import irene.pong.kayttoliittyma.PainallusAlas;
11  import irene.pong.kayttoliittyma.PainallusYlos;
12  import irene.pong.kayttoliittyma.PauseNappi;
13  import irene.pong.kayttoliittyma.PelinValinta;
14  import irene.pong.kayttoliittyma.VapautusAlas;
15  import irene.pong.kayttoliittyma.VapautusYlos;
16  import static javax.swing.JComponent.WHEN_IN_FOCUSED_WINDOW;
17  import javax.swing.KeyStroke;
18  import org.jdesktop.core.animation.timing.Animator;
19  import org.jdesktop.core.animation.timing.TimingSource;
20  import org.jdesktop.swing.animation.timing.sources.SwingTimerTimingSource;
21  
22  public class Pong { //ei yhtä luokkaa, joka luo sekä UI:n että logiikan elementtejä. Tulisi muuttaa logiikan pääluokaksi.
23  
24      Ikkuna ikkuna;
25      Animoija animoija;
26      Tilasto tilasto;
27      KomponenttiHallinta kontrolleri;
28      Pelityyppi pelityyppi;
29  
30      public Pong() {
31          ikkuna = new Ikkuna();
32          tilasto = new Tilasto(ikkuna.getKentta());
33          TimingSource ts = new SwingTimerTimingSource();
34          Animator.setDefaultTimingSource(ts);
35          ts.init();
36      }
37  
38      public Ikkuna getIkkuna() {
39          return ikkuna;
40      }
41  
42      public void setIkkuna(Ikkuna ikkuna) {
43          this.ikkuna = ikkuna;
44      }
45  
46      public Pelityyppi getPelityyppi() {
47          return pelityyppi;
48      }
49  
50      public Animoija getAnimoija() {
51          return animoija;
52      }
53  
54      public KomponenttiHallinta getKontrolleri() {
55          return kontrolleri;
56      }
57  
58      public void setPelityyppi(Pelityyppi pelityyppi) {
59          this.pelityyppi = pelityyppi;
60          lisaaPelaajienNappaimet(pelityyppi);
61          if (pelityyppi == Pelityyppi.KAKSINPELI) {
62              ikkuna.asetaIlmoitus("<html><div style='text-align: center;'> Valitsit kaksinpelin <br> Paina SPACE aloittaaksesi </div></html>");
63          } else if (pelityyppi == Pelityyppi.YKSINPELI) {
64              ikkuna.asetaIlmoitus("<html><div style='text-align: center;'> Valitsit yksinpelin (tekoäly) <br> Paina SPACE aloittaaksesi </div></html");
65          }
66      }
67  
68      public void aloita() {
69          ikkuna.alusta();
70          kontrolleri = new KomponenttiHallinta(ikkuna.getKentta(), tilasto);
71          kontrolleri.alustaKomponentit();
72          animoija = new Animoija(this);
73          lisaaKayttoLiittymaNappaimet();
74          ikkuna.asetaIlmoitus("<html><div style='text-align: center;'> Tervetuloa Pong-peliin!<br>Valitse pelityyppi: <br> 1 - yksinpeli <br> 2 - kaksinpeli </div></html>");
75  //        animoija.aloitaPeli();
76      }
77  
78      public void lisaaKayttoLiittymaNappaimet() {
79          ikkuna.getKentta().getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("pressed P"), "P painettu");
80          ikkuna.getKentta().getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("pressed SPACE"), "SPACE painettu");
81          ikkuna.getKentta().getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("pressed 1"), "1 painettu");
82          ikkuna.getKentta().getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("pressed 2"), "2 painettu");
83  
84          ikkuna.getKentta().getActionMap().put("P painettu", new PauseNappi(animoija, ikkuna));
85          ikkuna.getKentta().getActionMap().put("SPACE painettu", new AloitusPainallus(animoija, ikkuna));
86          ikkuna.getKentta().getActionMap().put("1 painettu", new PelinValinta(this, "1"));
87          ikkuna.getKentta().getActionMap().put("2 painettu", new PelinValinta(this, "2"));
88      }
89  
90      public void lisaaPelaajienNappaimet(Pelityyppi tyyppi) {
91  
92          ikkuna.getKentta().getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("pressed UP"), "UP painettu");
93          ikkuna.getKentta().getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("released UP"), "UP vapautettu");
94          ikkuna.getKentta().getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("pressed DOWN"), "DOWN painettu");
95          ikkuna.getKentta().getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("released DOWN"), "DOWN vapautettu");
96  
97          ikkuna.getKentta().getActionMap().put("UP painettu", new PainallusYlos(kontrolleri, "UP"));
98          ikkuna.getKentta().getActionMap().put("UP vapautettu", new VapautusYlos(kontrolleri, "UP"));
99          ikkuna.getKentta().getActionMap().put("DOWN painettu", new PainallusAlas(kontrolleri, "DOWN"));
100         ikkuna.getKentta().getActionMap().put("DOWN vapautettu", new VapautusAlas(kontrolleri, "DOWN"));
101 
102         if (tyyppi == Pelityyppi.KAKSINPELI) {
103             ikkuna.getKentta().getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("pressed W"), "W painettu");
104             ikkuna.getKentta().getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("released W"), "W vapautettu");
105             ikkuna.getKentta().getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("pressed S"), "S painettu");
106             ikkuna.getKentta().getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("released S"), "S vapautettu");
107 
108             ikkuna.getKentta().getActionMap().put("W painettu", new PainallusYlos(kontrolleri, "W"));
109             ikkuna.getKentta().getActionMap().put("W vapautettu", new VapautusYlos(kontrolleri, "W"));
110             ikkuna.getKentta().getActionMap().put("S painettu", new PainallusAlas(kontrolleri, "S"));
111             ikkuna.getKentta().getActionMap().put("S vapautettu", new VapautusAlas(kontrolleri, "S"));
112 
113         }
114 
115     }
