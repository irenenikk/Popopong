2   package irene.pong.logiikka;
3   
4   /**
5    * Luokka lisää pelaajille pisteitä ja tarkistaa voittajan.
6    * 
7    */
8   public class Tilasto {
9   
10      private final int voittoPisteet = 15;
11      private int vasenPisteet;
12      private int oikeaPisteet;
13      
14      /**
15       * Aluksi molempien pelaajien pisteet ovat nolla.
16       */
17      public Tilasto() {
18          vasenPisteet = 0;
19          oikeaPisteet = 0;
20      }
21      
22      /**
23       * Lisää pelaajalle pisteen.
24       * 
25       * @param p Pelaaja, jolle piste annetaan
26       */
27      public void lisaaPistePelaajalle(Pelaaja p) {
28          if (p == Pelaaja.VASEN) {
29              vasenPisteet++;
30          } else if (p == Pelaaja.OIKEA) {
31              oikeaPisteet++;
32          }
33      }
34      
35      /**
36       * Palauttaa vasemmanpuoleisen pelaajan pisteet.
37       * @return Vasemmanpuoleisen pelaajan pisteet.
38       */
39      public int vasemmanPelaajanPisteet() {
40          return vasenPisteet;
41      }
42      
43      /**
44       * Palauttaa oikeanpuoleisenpelaajan pisteet.
45       * @return Oikeanpuoleisen pelaajan pisteet.
46       */
47      public int oikeanPelaajanPisteet() {
48          return oikeaPisteet;
49      }    
50  
51      public void setVasenPisteet(int vasenPisteet) {
52          this.vasenPisteet = vasenPisteet;
53      }
54  
55      public void setOikeaPisteet(int oikeaPisteet) {
56          this.oikeaPisteet = oikeaPisteet;
57      }
58  
59      /**
60       * Vertailee pelaajien pisteitä keskenään ja palauttaa voittajan.
61       * 
62       * @return Voittanut pelaaja
63       */
64      public Pelaaja voittaja() {
65          if (vasenPisteet >= voittoPisteet) {
66              return Pelaaja.VASEN;
67          } else if (oikeaPisteet >= voittoPisteet) {
68              return Pelaaja.OIKEA;
69          } else {
70              return null;
71          }
72      }
73      
74      /**
75       * Nollaa pisteet. Kutsutaan, kun aloitetaan uusi peli.
76       */
77      public void nollaaPisteet() {
78          vasenPisteet = 0;
79          oikeaPisteet = 0;
80      }
81  }
