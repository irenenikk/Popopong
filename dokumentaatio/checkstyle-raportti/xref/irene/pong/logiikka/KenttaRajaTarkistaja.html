5   /**
6    * Luokka hallinnnoi Liikkuja-rajapinnan toteuttavien olioiden liikettä
7    * kentän reunoilla.
8    * 
9    */
10  public class KenttaRajaTarkistaja {
11      private Kentta kentta;
12  
13      public KenttaRajaTarkistaja(Kentta kentta) {
14          this.kentta = kentta;
15      }
16      
17      
18      /**
19       * Estää Liikkuva-rajapinnan toteuttavaa oliota poistumasta kentältä.
20       * 
21       * @param l Liikkuva-rajapinnan toteuttava maila.
22       */
23      public void tarkistaOsuukoKentanReunaan(Liikkuja l) {
24          if (l.getY() <= 0) {
25              l.setY(0);
26          }
27  
28          if (l.getY() + l.getKorkeus() >= kentta.getKorkeus()) {
29              l.setY(kentta.getKorkeus() - l.getKorkeus());
30          }
31      }
32  }
