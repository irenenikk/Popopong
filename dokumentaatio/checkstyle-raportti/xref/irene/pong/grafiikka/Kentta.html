1   /**
2    * Luokka toimii lähennä säiliönä pelin komponenteille, eli pallolle ja mailoille. 
3    * Kenttä myös toteuttaa varsinaisen osien maalaamisen.
4    */
5   
6   package irene.pong.grafiikka;
7   
8   import irene.pong.komponentit.Maila;
9   import irene.pong.komponentit.Pallo;
10  import irene.pong.komponentit.Pelaaja;
11  import java.awt.Color;
12  import java.awt.Font;
13  import java.awt.Graphics;
14  import java.util.List;
15  import javax.swing.JLabel;
16  import javax.swing.JPanel;
17  
18  public class Kentta extends JPanel { //ei ok, että luo Pallon ja Pelaajat --> logiikkaa. Tärkein työ maalaaminen
19      
20      private final Pallo pallo;
21      private Pelaaja pelaaja1;
22      private Pelaaja pelaaja2;
23      
24      public Kentta() {        
25          this.pallo = new Pallo();
26          this.pelaaja1 = new Pelaaja(1);
27          this.pelaaja2 = new Pelaaja(2); 
28      }
29  
30      public Pallo getPallo() {
31          return pallo;
32      }
33  
34      public Pelaaja getPelaaja1() {
35          return pelaaja1;
36      }
37  
38      public Pelaaja getPelaaja2() {
39          return pelaaja2;
40      }
41          
42      @Override
43      public void paintComponent(Graphics g) {
44          super.paintComponent(g);
45          
46          g.setColor(Color.LIGHT_GRAY);
47          g.setFont(new Font("Arial", Font.BOLD, 50)); 
48          g.drawString(Integer.toString(pelaaja1.getPisteet()), 50, 50);
49          g.drawString(Integer.toString(pelaaja2.getPisteet()), getWidth() - 100, 50);
50          
51          g.setColor(Color.WHITE);
52  
53          g.fillOval(pallo.getX(), pallo.getY(), pallo.getHalkaisija(), pallo.getHalkaisija());
54          g.fillRect(pelaaja1.getMaila().getX(), pelaaja1.getMaila().getY(), pelaaja1.getMaila().getLeveys(), pelaaja1.getMaila().getKorkeus());
55          g.fillRect(pelaaja2.getMaila().getX(), pelaaja2.getMaila().getY(), pelaaja2.getMaila().getLeveys(), pelaaja2.getMaila().getKorkeus());
56          
57          getToolkit().sync();
58      }
59  }
