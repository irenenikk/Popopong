1   /**
2    * Luokka piirtää pelikentän sisällön.
3    */
4   package irene.pong.kayttoliittyma;
5   
6   import irene.pong.komponentit.Este;
7   import irene.pong.logiikka.Paivitettava;
8   import irene.pong.logiikka.Pong;
9   import java.awt.Color;
10  import java.awt.Font;
11  import java.awt.Graphics;
12  import java.util.*;
13  import javax.swing.JPanel;
14  
15  public class Piirturi extends JPanel implements Paivitettava {
16  
17      private final Pong logiikka;
18      private HashMap<Este, Color> esteidenVarit;
19      private final Random r;
20  
21      /**
22       * @param pong peli, jonka kentän sisältö ja pisteet luokan tulee piirtää.
23       */
24      public Piirturi(Pong pong) {
25          logiikka = pong;
26          r = new Random();
27          esteidenVarit = new HashMap();
28      }
29  
30      public void tyhjennaEsteidenVarit() {
31          esteidenVarit.clear();
32      }
33  
34      @Override
35      public void paintComponent(Graphics g) {
36          super.paintComponent(g);
37  
38          g.setColor(Color.LIGHT_GRAY);
39  
40          g.setFont(new Font("Arial", Font.BOLD, 50));
41          g.drawString(Integer.toString(logiikka.getTilasto().vasemmanPelaajanPisteet()), 50, 50);
42          g.drawString(Integer.toString(logiikka.getTilasto().oikeanPelaajanPisteet()), getWidth() - 100, 50);
43  
44          if (logiikka.getKentta().getEsteet().size() > 0) {
45              logiikka.getKentta().getEsteet().stream().forEach((este) -> {
46                  if (!este.isNakyvissa()) {
47                      g.setColor(Color.DARK_GRAY);
48                  } else {
49                      if (!esteidenVarit.containsKey(este)) {
50                          esteidenVarit.put(este, new Color(r.nextFloat(), r.nextFloat(), r.nextFloat()));
51                      }
52                      g.setColor(esteidenVarit.get(este));
53                  }
54                  g.fillRect(este.getX(), este.getY(), este.getLeveys(), este.getKorkeus());
55              });
56          }
57  
58          g.setColor(Color.WHITE);
59  
60          g.fillRect(logiikka.getKentta().getVasenPelaaja().getX(), logiikka.getKentta().getVasenPelaaja().getY(), logiikka.getKentta().getVasenPelaaja().getLeveys(), logiikka.getKentta().getVasenPelaaja().getKorkeus());
61          g.fillRect(logiikka.getKentta().getOikeaPelaaja().getX(), logiikka.getKentta().getOikeaPelaaja().getY(), logiikka.getKentta().getOikeaPelaaja().getLeveys(), logiikka.getKentta().getOikeaPelaaja().getKorkeus());
62  
63          g.fillOval(logiikka.getKentta().getPallo().getX(), logiikka.getKentta().getPallo().getY(), logiikka.getKentta().getPallo().getHalkaisija(), logiikka.getKentta().getPallo().getHalkaisija());
64  
65          getToolkit().sync();
66      }
67  
68      @Override
69      public void paivita() {
70          this.repaint();
71      }
72  
73  }
