4   
5   package irene.pong.kayttoliittyma;
6   
7   import irene.pong.logiikka.Paivitettava;
8   import irene.pong.logiikka.Pong;
9   import java.awt.Color;
10  import java.awt.Font;
11  import java.awt.Graphics;
12  import javax.swing.JPanel;
13  
14  public class Piirturi extends JPanel implements Paivitettava {
15      private final Pong logiikka;
16      
17      public Piirturi(Pong pong) {
18          /**
19           * @param pong peli, jonka kentän sisältö ja pisteet luokan tulee piirtää.
20           */
21          logiikka = pong;
22      }
23      
24      @Override
25      public void paintComponent(Graphics g) {
26          super.paintComponent(g);
27         
28          g.setColor(Color.LIGHT_GRAY);
29          g.setFont(new Font("Arial", Font.BOLD, 50)); 
30          g.drawString(Integer.toString(logiikka.getTilasto().vasemmanPelaajanPisteet()), 50, 50);
31          g.drawString(Integer.toString(logiikka.getTilasto().oikeanPelaajanPisteet()), getWidth() - 100, 50);
32          
33          g.setColor(Color.WHITE);
34  
35          g.fillOval(logiikka.getKentta().getPallo().getX(), logiikka.getKentta().getPallo().getY(), logiikka.getKentta().getPallo().getHalkaisija(), logiikka.getKentta().getPallo().getHalkaisija());
36          g.fillRect(logiikka.getKentta().getVasenPelaaja().getX(), logiikka.getKentta().getVasenPelaaja().getY(), logiikka.getKentta().getVasenPelaaja().getLeveys(), logiikka.getKentta().getVasenPelaaja().getKorkeus());
37          g.fillRect(logiikka.getKentta().getOikeaPelaaja().getX(), logiikka.getKentta().getOikeaPelaaja().getY(), logiikka.getKentta().getOikeaPelaaja().getLeveys(), logiikka.getKentta().getOikeaPelaaja().getKorkeus());
38          
39          getToolkit().sync();
40      }    
41  
42      @Override
43      public void paivita() {
44          this.repaint();
45      }
46      
47  }
