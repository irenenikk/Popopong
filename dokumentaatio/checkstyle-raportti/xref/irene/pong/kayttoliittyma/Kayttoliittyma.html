23  import static javax.swing.JComponent.WHEN_IN_FOCUSED_WINDOW;
24  import javax.swing.JFrame;

31  
32  public class Kayttoliittyma implements Runnable, Paivitettava { //hyvä lähtökohta pääUI-luokalle
33  
34      private JFrame ikkuna;
35  //    private final Kentta kentta;
36      private JLabel ilmoitus;
37      private JLabel pause;
38      private Pong logiikka;
39      private Piirturi piirturi;
40      private final Animoija animoija;
41      
42  
43      public Kayttoliittyma(Pong pong) {
44          /**
45           * Luo kaikki käyttöliittymän komponentit, ja asettaa itsensä päälogiikkaluokan päivitettävä-olioksi.
46           * 
47           * @param pong päälogiikkaluokka Pong
48           * 
49           * @see Pong
50           * @see Piirturi 
51           */
52          TimingSource ts = new SwingTimerTimingSource();
53          Animator.setDefaultTimingSource(ts);
54          ts.init();
55          logiikka = pong;
56          ikkuna = new JFrame();
57          piirturi = new Piirturi(pong);
58          piirturi.setLayout(new BorderLayout());
59          ilmoitus = new JLabel();
60          pause = new JLabel("p - pause");
61          animoija = new Animoija(logiikka);
62          logiikka.setLiikePaivitettava(piirturi);
63          logiikka.setTilannePaivitettava(this);
64      }
65      
66      public Pong getLogiikka() {
67          return logiikka;
68      }
69      
70      public Animoija getAnimoija() {
71          return animoija;
72      }
73      
74      public void asetaIlmoitus(String ilmo) {
75          ilmoitus.setText(ilmo);
76      }
77      
78      public void tyhjennaIlmoitus() {
79          ilmoitus.setText("");
80      }
81      
82      public void pauseNakyvaksi() {
83          pause.setVisible(true);
84      }
85      
86      public void pausePiiloon() {
87          pause.setVisible(false);
88      }
89      
90      public void alustaIkkuna() {
91          ikkuna.setPreferredSize(new Dimension(logiikka.getKentta().getLeveys(), logiikka.getKentta().getKorkeus()));
92          ikkuna.setTitle("Pong");
93          ikkuna.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
94          ikkuna.setResizable(false);        
95      }
96      
97      public void alustaIlmoitus() {        
98          ilmoitus.setFont(new Font("", Font.PLAIN, 20)); //Fontin nimen muuttaminen ei vaikuta
99          ilmoitus.setHorizontalAlignment(SwingConstants.CENTER);  
100         ilmoitus.setVerticalAlignment(SwingConstants.NORTH);
101         ilmoitus.setForeground(Color.WHITE);
102         
103     }
104     
105     public void alustaPauseNappi() {
106         pause.setHorizontalAlignment(SwingConstants.RIGHT);
107         pause.setForeground(Color.WHITE);  
108         pause.setVisible(false);
109     }  
110     
111     public void alustaPiirturi() {
112         piirturi.setPreferredSize(new Dimension(logiikka.getKentta().getLeveys(), logiikka.getKentta().getKorkeus()));
113         piirturi.setBackground(Color.BLACK);
114         piirturi.add(ilmoitus, BorderLayout.CENTER);
115         piirturi.add(pause, BorderLayout.SOUTH);
116         ikkuna.add(piirturi);        
117     }
118 
119     public void lisaaKayttoLiittymaNappaimet() {
120         piirturi.getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("pressed P"), "P painettu");
121         piirturi.getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("pressed SPACE"), "SPACE painettu");
122         piirturi.getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("pressed 1"), "1 painettu");
123         piirturi.getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("pressed 2"), "2 painettu");
124 
125         piirturi.getActionMap().put("P painettu", new PauseNappi(logiikka, this));
126         piirturi.getActionMap().put("SPACE painettu", new AloitusPainallus(logiikka, this));
127         piirturi.getActionMap().put("1 painettu", new PelinValinta(logiikka, "1", this));
128         piirturi.getActionMap().put("2 painettu", new PelinValinta(logiikka, "2", this));
129     }
130 
131     public void lisaaPelaajienNappaimet(Pelityyppi tyyppi) {
132 
133         piirturi.getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("pressed UP"), "UP painettu");
134         piirturi.getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("released UP"), "UP vapautettu");
135         piirturi.getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("pressed DOWN"), "DOWN painettu");
136         piirturi.getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("released DOWN"), "DOWN vapautettu");
137 
138         piirturi.getActionMap().put("UP painettu", new PainallusYlos(logiikka.getKontrolleri(), "UP"));
139         piirturi.getActionMap().put("UP vapautettu", new VapautusYlos(logiikka.getKontrolleri(), "UP"));
140         piirturi.getActionMap().put("DOWN painettu", new PainallusAlas(logiikka.getKontrolleri(), "DOWN"));
141         piirturi.getActionMap().put("DOWN vapautettu", new VapautusAlas(logiikka.getKontrolleri(), "DOWN"));
142 
143         if (tyyppi == Pelityyppi.KAKSINPELI) {
144             piirturi.getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("pressed W"), "W painettu");
145             piirturi.getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("released W"), "W vapautettu");
146             piirturi.getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("pressed S"), "S painettu");
147             piirturi.getInputMap(WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke("released S"), "S vapautettu");
148 
149             piirturi.getActionMap().put("W painettu", new PainallusYlos(logiikka.getKontrolleri(), "W"));
150             piirturi.getActionMap().put("W vapautettu", new VapautusYlos(logiikka.getKontrolleri(), "W"));
151             piirturi.getActionMap().put("S painettu", new PainallusAlas(logiikka.getKontrolleri(), "S"));
152             piirturi.getActionMap().put("S vapautettu", new VapautusAlas(logiikka.getKontrolleri(), "S"));
153 
154         }
155 
156     }

158     @Override
159     public void run() {
160         alustaIkkuna();
161         alustaIlmoitus();
162         alustaPauseNappi();
163         alustaPiirturi();       
164         ikkuna.pack();
165         ikkuna.setVisible(true);
166         
167         lisaaKayttoLiittymaNappaimet();
168 
169         logiikka.aloita();
170     }

172     @Override
173     public void paivita() {
174         
175         if (logiikka.getTilasto().voittaja() != null && !logiikka.isPeliKaynnissa() && logiikka.peliAlkanut()) {
176             String voittaja = "";
177             if (logiikka.getTilasto().voittaja() == Pelaaja.VASEN && logiikka.getPelityyppi() == Pelityyppi.YKSINPELI) {
178                 voittaja = "AI";
179             } else if (logiikka.getTilasto().voittaja() == Pelaaja.VASEN && logiikka.getPelityyppi() == Pelityyppi.KAKSINPELI) {
180                 voittaja = "Player 1";
181             } else if (logiikka.getTilasto().voittaja() == Pelaaja.OIKEA) {
182                 voittaja = "Player 2";
183             }
184             asetaIlmoitus("<html><div style='text-align: center;'> <strong> " + voittaja + " wins </strong> <br> Press SPACE to restart <br> or close the window </div></html>");
185             animoija.getAnimoija().stop();
186             
187         } else if (!logiikka.peliAlkanut() && !logiikka.isPeliKaynnissa() && logiikka.getPelityyppi() != null) {
188             String pelityyppi = "";
189             String ohjeet = "";
190             if (logiikka.getPelityyppi() == Pelityyppi.YKSINPELI) {
191                 pelityyppi = "single"; 
192                 ohjeet = "Use the arrow keys";
193             } else if (logiikka.getPelityyppi() == Pelityyppi.KAKSINPELI) {
194                 pelityyppi = "multiplayer";
195                 ohjeet = "Player 1: W - S <br> Player 2: Arrow keys";
196             }
197             asetaIlmoitus("<html><div style='text-align: center;'> <p> You chose " + pelityyppi + "</p> <p>  " + ohjeet + " </p> <p> Press SPACE to start </p></div></html>");
198         } else if (!logiikka.peliAlkanut() && !logiikka.isPeliKaynnissa() && logiikka.getPelityyppi() == null && logiikka.getTilasto().voittaja() == null) {
199             asetaIlmoitus("<html><div style='text-align: center;'> <strong> PONG </strong> <p> 1 - single <br> 2 - multiplayer </p> </div></html>");
200         }
201         
202     }
