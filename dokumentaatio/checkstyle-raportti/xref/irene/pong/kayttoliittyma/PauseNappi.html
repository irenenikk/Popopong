12  public class PauseNappi extends AbstractAction {
13  
14      private final Animoija animoija;
15      private final Ikkuna ikkuna;
16      
17      public PauseNappi (Animoija a, Ikkuna i) {
18          animoija = a;
19          ikkuna = i;
20      }
21  
22      @Override
23      public void actionPerformed(ActionEvent e) {
24          if (animoija.isPeliKaynnissa()) {
25              animoija.setPeliKaynnissa(false);
26              ikkuna.asetaIlmoitus("PYSÃ„YTETTY");
27              ikkuna.pausePiiloon();
28              animoija.pauseta();
29          } else {
30              animoija.setPeliKaynnissa(true);
31              ikkuna.asetaIlmoitus("");
32              ikkuna.pauseNakyvaksi();
33              animoija.jatka();
34          }
35      }
36  }
